-- Criar as tabelas
CREATE TABLE AUTOR (
    ID_AUTOR SERIAL PRIMARY KEY,
    NOME VARCHAR(100),
    CPF VARCHAR(14),
    EMAIL VARCHAR(100)
);

CREATE TABLE LIVROS (
    ID_LIVRO SERIAL PRIMARY KEY,
    ID_AUTOR INT,
    TITULO VARCHAR(255),
    PRECO NUMERIC(10,2),
    RESUMO TEXT,
    FAIXA_ETARIA CHAR(1),
    FOREIGN KEY (ID_AUTOR) REFERENCES AUTOR(ID_AUTOR)
);

CREATE TABLE LIVRARIA (
    ID_LIVRARIA SERIAL PRIMARY KEY,
    NOME VARCHAR(255),
    ENDERECO VARCHAR(255),
    CNPJ VARCHAR(18)
);

CREATE TABLE CLIENTE (
    ID_CLIENTE SERIAL PRIMARY KEY,
    NOME VARCHAR(100),
    CPF VARCHAR(14),
    TELEFONE VARCHAR(20)
);

CREATE TABLE VENDEDOR (
    ID_VENDEDOR SERIAL PRIMARY KEY,
    NOME VARCHAR(100),
    CPF VARCHAR(14),
    FUNCAO CHAR(1),
    ADMISSAO DATE
);

CREATE TABLE VENDA (
    ID_VENDA SERIAL PRIMARY KEY,
    ID_LIVRARIA INT,
    ID_LIVRO INT,
    ID_VENDEDOR INT,
    ID_CLIENTE INT,
    QUANTIDADE INT,
    DATA DATE,
    FORMA_DE_PAGAMENTO CHAR(1),
    FOREIGN KEY (ID_LIVRARIA) REFERENCES LIVRARIA(ID_LIVRARIA),fai
    FOREIGN KEY (ID_LIVRO) REFERENCES LIVROS(ID_LIVRO),
    FOREIGN KEY (ID_VENDEDOR) REFERENCES VENDEDOR(ID_VENDEDOR),
    FOREIGN KEY (ID_CLIENTE) REFERENCES CLIENTE(ID_CLIENTE)
);
